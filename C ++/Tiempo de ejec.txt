float fuzzy(double X,double A,double B,double C) {
    float m=0.0;
    if ((A<=X) && (X<=B)) {
        m=(X-A)/(B-A);
    } else if ((B<=X) && (X<=C)) {
        m=(C-X)/(C-B);
    } else {
        m=0.0;
    }
    return m;
}

}
